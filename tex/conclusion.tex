\begin{frame} %-----------------------------%
\frametitle{Conclusions}
  \begin{enumerate}
    \item<1-> Systematic method of designing low-thrust orbital transfers via \Emph{reachability sets}
    \begin{itemize}
        \item Alleviates the need for selecting accurate initial guesses 
        \item Gives insight into the possible motion of the spacecraft
    \end{itemize}
    \item<2-> \Emph{Constrained attitude control} allows for simple method of satisfying attitude constraints
    \begin{itemize}
        \item Completely avoids singularities and ambiguities
        \item Geometrically exact and computationally efficient
    \end{itemize}
    \item<3-> \Emph{Asteroid landing} requires the coupled approach 
    \begin{itemize}
        \item Global mission scenarios are possible
        \item Dynamics derived in a consistent and geometrically exact framework
        \item \Emph{Variational Integrators} provide means of accurate long term propogation
    \end{itemize}

  \end{enumerate}
  \note[itemize]{
  \item asteroid landing requires techniques developed in both of the previous two bullets
  }
\end{frame}   %-----------------------------%
